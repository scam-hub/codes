#install ftp&telnet in client
sudo apt install ftp
sudo apt install telnet

------------------------ FTP commands ------------------------
sudo apt-get install vsftpd

sudo nano /etc/vsftpf.conf

In this file add,
write_enable=YES
and ensure that,
anonymous_enable=NO

sudo systemctl restart vsftpd (If any error occurs after this command, then type: sudo systemctl daemon-reload)

sudo systemctl status vsftpd

ftp <hostname>

#some commands:
get <filename>
put <filename>

---------------------- Telnet commands ----------------------

sudo apt install xinetd telnetd -y

sudo nano /etc/xinetd.d/telnet

In this file add,
service telnet
{
  disable = no
  socket_type = stream
  wait = no
  user = root
  server = /usr/sbin/telnetd
}

sudo systemctl restart xinetd

sudo systemctl enable xinetd

sudo systemctl status xinetd

sudo ufw allow 23/tcp

telnet <hostname>
